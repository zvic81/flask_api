This index page flask_api server

Use terminal and curl for access api server or use Postman

GET curl --location --request GET 'http://127.0.0.1:5000/goods' \ for all goods

GET curl --location --request GET 'http://127.0.0.1:5000/goods/3' \ for goods id=3

POST curl --location --request POST 'http://127.0.0.1:5000/goods' \ --header 'Content-Type: application/json' \ --data-raw '{ "name":"oranges", "price":8, "manufacture_date":"2/11/11", "picture_url":"gogle.com" }' for append goods with json

PUT curl --location --request PUT 'http://127.0.0.1:5000/goods/17' \ --header 'Content-Type: application/json' \ --data-raw '{ "name":"candy", "price":8, "manufacture_date":"2/11/11", "picture_url":"gogle.com" }' for change goods id=3 with json

DELETE curl --location --request DELETE 'http://127.0.0.1:5000/goods/17' \ --header 'Content-Type: application/json' \ --data-raw '{ "name":"oranges", "price":8, "manufacture_date":"2/11/11", "picture_url":"gogle.com" }' for delete goods id=3

format json: { "name":"bananas", "price":8, "manufacture_date":"2/11/11", "picture_url":"gogle.com" }
